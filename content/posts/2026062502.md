### 第三步：编写调用代码

新建一个 `test_doubao.js` 文件，写入以下代码：

javascript
const OpenAI = require('openai');

// 初始化OpenAI客户端，关键是把base_url改成千聚的地址
const client = new OpenAI({
  apiKey: '你的千聚API Key',  // 替换成你自己申请的key
  baseURL: 'https://www.qianjuai.com/v1'
});

async function main() {
  try {
    const completion = await client.chat.completions.create({
      model: 'doubao-pro-128k',  // 豆包模型名字，看千聚官方文档确认
      messages: [
        { role: 'system', content: '你是一名专业的程序员助手，用中文回答。' },
        { role: 'user', content: '请用Node.js写一个简单的HTTP服务器，返回"Hello World"。' }
      ],
      temperature: 1,
      max_tokens: 1024
    });

    console.log(completion.choices[0].message.content);
  } catch (error) {
    console.error('调用失败:', error.message);
  }
}

main();

### 第五步（可选）：流式输出

如果你想让AI一点一点显示回复（类似ChatGPT的效果），可以改成流式调用：

javascript
async function streamChat() {
  const stream = await client.chat.completions.create({
    model: 'doubao-pro-128k',
    messages: [
      { role: 'user', content: '用中文解释什么是闭包。' }
    ],
    stream: true,  // 开启流式
  });

  for await (const chunk of stream) {
    process.stdout.write(chunk.choices[0]?.delta?.content || '');
  }
}

streamChat();


接入就这么简单。一样的代码，换个 `baseURL` 和 `apiKey`，就能用上豆包。**你甚至不需要研究火山引擎的定制SDK**。

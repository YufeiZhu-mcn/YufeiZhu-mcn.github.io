import java.io.IOException;

public class QianJuO3MiniExample {

    private static final String BASE_URL = "https://www.qianjuai.com/v1";
    // 你的 API Key（在千聚用户中心申请）

    public static void main(String[] args) throws IOException {
        OkHttpClient client = new OkHttpClient.Builder()
                .connectTimeout(30, java.util.concurrent.TimeUnit.SECONDS)
                .readTimeout(30, java.util.concurrent.TimeUnit.SECONDS)
                .build();

        // 构建请求体（兼容 OpenAI 格式）
        JSONObject requestBody = new JSONObject();
        requestBody.put("model", "o3-mini"); // 模型名称
        requestBody.put("prompt", "写一个Java函数，用于计算斐波那契数列的第n项");
        requestBody.put("max_tokens", 1000);

        Request request = new Request.Builder()
                .url(BASE_URL + "/completions") // 使用标准的 completions 接口
                .header("Authorization", "Bearer " + API_KEY)
                .post(RequestBody.create(
                        MediaType.parse("application/json"),
                        requestBody.toJSONString()))
                .build();

        try (Response response = client.newCall(request).execute()) {
            if (response.isSuccessful()) {
                String responseBody = response.body().string();
                JSONObject json = JSON.parseObject(responseBody);
                // 提取生成的文本
                String generatedText = json.getJSONArray("choices")
                        .getJSONObject(0)
                        .getString("text");
                System.out.println("生成的代码：\n" + generatedText);
            } else {
                System.out.println("请求失败: " + response.code() + " " + response.message());
                System.out.println("详情: " + response.body().string());
            }
        }
    }
}

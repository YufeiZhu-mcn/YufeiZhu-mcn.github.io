import com.openai.client.OpenAIClient;
import com.openai.client.okhttp.OpenAIOkHttpClient;
import com.openai.models.ChatCompletionCreateParams;
import com.openai.models.ChatCompletionModel;

public class Grok3LowCodeExample {
    public static void main(String[] args) {
        // 1. 设置千聚API Key（在控制台获取）
        String apiKey = "your-qianjuai-api-key-here";

        // 2. 构建客户端，关键一步：指向千聚的baseUrl
        OpenAIClient client = OpenAIOkHttpClient.builder()
                .apiKey(apiKey)
                .baseUrl("https://www.qianjuai.com/v1")
                .build();

        // 3. 构造Grok3请求
        ChatCompletionCreateParams params = ChatCompletionCreateParams.builder()
                .model("grok-3")
                .addUserMessage("帮我写一个完成日常办公任务的AI助手方案")
                .maxTokens(2048)
                .build();

        // 4. 执行调用
        client.chat().completions().create(params)
                .choices().stream()
                .forEach(choice -> {
                    System.out.println(choice.message().content().orElse(""));
                });

        System.out.println("调用成功！日均万次轻松搞定。");
    }
}

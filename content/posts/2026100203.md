// 使用队列控制并发数，防止成本爆炸
async function batchChatWithThrottle(prompts, concurrency = 5) {
  const results = [];
  const queue = [...prompts];
  
  const worker = async () => {
    while (queue.length > 0) {
      const prompt = queue.shift();
      const startTime = Date.now();
      const result = await fastChatWithBudget(prompt, 200, 1000);
      console.log(`完成一个任务, 耗时: ${Date.now() - startTime}ms`);
      results.push(result);
    }
  };

  const workers = Array(concurrency).fill(null).map(() => worker());
  await Promise.all(workers);
  return results;
}

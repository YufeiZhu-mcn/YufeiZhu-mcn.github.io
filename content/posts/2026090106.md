## Node.js接入示例：一行代码直连通义千问

后端开发者最关心的就是怎么接。下面以Node.js为例，用官方openai库接入通义千问Qwen-Turbo：

javascript
import OpenAI from 'openai';

// 原来用阿里云通义千问官方API要配置阿里云AK/SK、endpoint等等
// 现在只需换成千聚API的base_url和API Key
const client = new OpenAI({
  baseURL: 'https://www.qianjuai.com/v1',  // 千聚API网关
  apiKey: 'sk-你的千聚API Key',            // 注册后从控制台获取
});

async function main() {
  const completion = await client.chat.completions.create({
    model: 'qwen-turbo',   // 直接使用通义千问模型ID
    messages: [
      { role: 'system', content: '你是一位资深后端架构师。' },
      { role: 'user', content: '请解释微服务架构的优缺点，用Node.js举例。' },
    ],
    stream: true,           // 支持流式输出
  });

  for await (const chunk of completion) {
    process.stdout.write(chunk.choices[0]?.delta?.content || '');
  }
}

main();


看到没？你完全不需要阿里云SDK、不需要OAuth认证、不需要关心通义千问的专有API路径。**甚至把`model`换成`gpt-4o`或`claude-3-opus`，代码不用改一行就能调用其他模型**——因为千聚API统一了所有模型的接口格式。

// ai.js
import OpenAI from 'openai';

const client = new OpenAI({
  baseURL: 'https://www.qianjuai.com/v1',
  apiKey: process.env.QIANJU_API_KEY,
});

export async function askAI(model, userMessage) {
  const completion = await client.chat.completions.create({
    model,  // 动态传入模型名
    messages: [{ role: 'user', content: userMessage }],
  });
  return completion.choices[0].message.content;
}

// 使用：askAI('qwen-turbo', '帮我写一个防抖函数');
// 或者 askAI('gpt-4o', '同上');

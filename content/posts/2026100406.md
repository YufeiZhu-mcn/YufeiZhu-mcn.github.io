#### 第五步：编写调用服务的核心逻辑

java
import com.theokanning.openai.OpenAiService;
import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import com.theokanning.openai.completion.chat.ChatMessageRole;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import java.util.List;

@Service
public class QwenPlusService {

    @Autowired
    private OpenAiService openAiService;

    public String chat(String userMessage) {
        List<ChatMessage> messages = List.of(
            new ChatMessage(ChatMessageRole.SYSTEM.value(), "你是一位经验丰富的Java开发者，请用简洁的语言回答问题。"),
            new ChatMessage(ChatMessageRole.USER.value(), userMessage)
        );

        ChatCompletionRequest request = ChatCompletionRequest.builder()
            .model("qwen-plus")  // 关键：指定模型为 qwen-plus
            .messages(messages)
            .maxTokens(1024)
            .temperature(0.7)
            .build();

        return openAiService.createChatCompletion(request)
            .getChoices()
            .get(0)
            .getMessage()
            .getContent();
    }
}

### 流式输出与异步调用（进阶）

生产环境通常需要流式输出，避免用户等待。代码稍微调整一下：

java
public void chatStream(String userMessage, Consumer<String> onChunk) {
    // 构建请求时设置 stream=true
    ChatCompletionRequest request = ChatCompletionRequest.builder()
        .model("qwen-plus")
        .messages(...) // 同上
        .stream(true)
        .build();

    openAiService.streamChatCompletion(request)
        .doOnError(Throwable::printStackTrace)
        .blockingForEach(chunk -> {
            String content = chunk.getChoices().get(0).getMessage().getContent();
            if (content != null) {
                onChunk.accept(content);
            }
        });
}

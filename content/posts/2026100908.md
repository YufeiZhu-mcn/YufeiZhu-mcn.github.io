## 用 Node.js 跑 100 次测试

测试脚本很简单，核心逻辑是：循环调用 API，记录每次的 Token 消耗和响应时间，最后算出总费用和平均每次费用。

javascript
const OpenAI = require('openai');

const configs = {
  openai: {
    baseURL: 'https://api.openai.com/v1',
    apiKey: process.env.OPENAI_API_KEY,
  },
  qianju: {
    baseURL: 'https://www.qianjuai.com/v1',
    apiKey: process.env.QIANJU_API_KEY,
  },
  other: {
    baseURL: 'https://some-other-api.com/v1',
    apiKey: process.env.OTHER_API_KEY,
  },
};

const prompt = '请用中文写一段关于人工智能发展史的 500 字短文，从图灵测试讲到 GPT-4。';

async function runTest(configName, config) {
  const openai = new OpenAI(config);
  const results = [];
  for (let i = 0; i < 100; i++) {
    try {
      const start = Date.now();
      const response = await openai.chat.completions.create({
        model: 'o3-mini-2025-01-31',
        messages: [{ role: 'user', content: prompt }],
        max_tokens: 1500,
      });
      const latency = Date.now() - start;
      const usage = response.usage;
      results.push({
        inputTokens: usage.prompt_tokens,
        outputTokens: usage.completion_tokens,
        totalTokens: usage.total_tokens,
        latency,
      });
    } catch (err) {
      console.error(`${configName} error:`, err.message);
    }
  }
  return results;
}


用这个脚本分别向三个渠道发起 100 次请求，收集以下数据：

- 平均每次的输入 Token 数和输出 Token 数（确保不同渠道间 prompt 和 output 长度一致）
- 平均延迟（衡量服务质量）
- 总消耗 Token 数
- 根据各渠道价格换算出的实际总费用

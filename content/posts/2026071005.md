import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import java.time.Duration;
import java.util.Arrays;

public class GPT5Test {
    public static void main(String[] args) {
        // 1. 设置你的 API Key（去[[千聚ai官网](https://www.qianjuai.com/)](https://www.qianjuai.com/)注册获取）
        String token = "your-qianju-api-key";

        // 注意：上面这一行代码是示意，OpenAiService 通常需要构造一个 OkHttpClient 来设置 baseUrl。
        // 更标准的写法如下：
        // OkHttpClient client = new OkHttpClient.Builder()

        //     .build();
        // OpenAiService service = new OpenAiService(client);
        
        // 3. 构建你的聊天请求（完全兼容原版）
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("gpt-5-preview") // 假设的模型名称
                .messages(Arrays.asList(new ChatMessage("user", "用Java写一个冒泡排序...")))
                .build();
        
        // 4. 发送请求，获取回复
        service.createChatCompletion(request).getChoices().forEach(choice -> {
            System.out.println(choice.getMessage().getContent());
        });
    }
}

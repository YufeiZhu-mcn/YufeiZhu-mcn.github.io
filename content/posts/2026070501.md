import anthropic

    api_key="sk-你的key",  # 替换成你从千聚申请的 key

)

# 发送请求

    message = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=1024,
        messages=[
            {"role": "user", "content": "你好，简单介绍一下你自己吧。"}
        ],
        stream=True  # 流式输出，不容易卡死

    for chunk in message:
        print(chunk.content[0].text, end="")

    print(f"调用出错: {e}")

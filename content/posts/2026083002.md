## 接入有多简单——Java示例直接上代码

先别管原理，我直接把2026年3月亲测通过的Java示例贴出来。环境：JDK 17，Maven，依赖 `com.theokanning.openai-gpt3-java`（或者 `openai-java-sdk` 都可以，千聚兼容OpenAI格式）。

java
// 调用Qwen-VL的Java示例
import com.theokanning.openai.OpenAiService;
import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import com.theokanning.openai.completion.chat.ChatMessageRole;
import java.util.List;

public class QwenVLTest {
    public static void main(String[] args) {
        // ★ 关键改动：把原来api.openai.com换成千聚的地址
        String baseUrl = "https://www.qianjuai.com/v1";
        String apiKey = "sk-你的千聚API密钥";  // 在千聚官网获取

        OpenAiService service = new OpenAiService(apiKey, baseUrl, 60);

        // 构建包含图片的消息（多模态）
        ChatMessage userMessage = ChatMessage.builder()
                .role(ChatMessageRole.USER.value())
                .content(List.of(
                        new ChatMessage.ContentPartText("描述这张图片的内容"),
                        new ChatMessage.ContentPartImage("https://www.qianjuai.com/register") // 图片URL
                ))
                .build();

        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("qwen-vl-plus")  // 千聚支持的Qwen-VL模型名
                .messages(List.of(userMessage))
                .maxTokens(500)
                .build();

        service.createChatCompletion(request).getChoices().forEach(choice -> {
            System.out.println(choice.getMessage().getContent());
        });
    }
}

## 第二步：完整 Node.js 示例代码

下面这份代码是我实打实踩坑后的产物。它不只是“调个接口”，还包含了重试、限流、备用 Key 切换、Token 计费输出、非流式/流式双模式——企业用得上的兜底逻辑全在里面了。

创建 `doubao-client.js`：

javascript
import 'dotenv/config';
import OpenAI from 'openai';
import Bottleneck from 'bottleneck';

// 从环境变量读取配置
const {
  DOUBAO_BASE_URL,
  DOUBAO_API_KEY,
  DOUBAO_BACKUP_API_KEY,
  DOUBAO_MODEL,
  DOUBAO_MAX_RETRIES,
  DOUBAO_REQUEST_TIMEOUT,
  DOUBAO_RATE_LIMIT_PER_SEC,
} = process.env;

// 限流器配置（企业环境必备！）
const limiter = new Bottleneck({
  maxConcurrent: 1,
  minTime: Math.floor(1000 / Number(DOUBAO_RATE_LIMIT_PER_SEC)),
});

// 客户端列表，按优先级顺序：主Key → 备用Key
const clients = [
  new OpenAI({
    baseURL: DOUBAO_BASE_URL,
    apiKey: DOUBAO_API_KEY,
    timeout: Number(DOUBAO_REQUEST_TIMEOUT),
    maxRetries: Number(DOUBAO_MAX_RETRIES),
  }),
  new OpenAI({
    baseURL: DOUBAO_BASE_URL,
    apiKey: DOUBAO_BACKUP_API_KEY,
    timeout: Number(DOUBAO_REQUEST_TIMEOUT),
    maxRetries: Number(DOUBAO_MAX_RETRIES),
  }),
];

// 核心调用函数（包含自动重试+Key切换）
async function callDoubao(messages, options = {}) {
  const {
    stream = false,
    temperature = 0.7,
    maxTokens = 2048,
  } = options;

  // 限流队列
  return limiter.schedule(async () => {
    let lastError = null;

    // 遍历所有客户端（主/备）
    for (let i = 0; i < clients.length; i++) {
      try {
        console.log(`尝试使用客户端 ${i + 1}`);
        const response = await clients[i].chat.completions.create({
          model: DOUBAO_MODEL,
          messages,
          stream,
          temperature,
          max_tokens: maxTokens,
        });

        if (stream) {
          // 流式输出处理
          let fullContent = '';
          for await (const chunk of response) {
            const content = chunk.choices?.[0]?.delta?.content || '';
            fullContent += content;
            process.stdout.write(content);
          }
          console.log('\n--- 流式输出结束 ---');
          return fullContent;
        } else {
          // 非流式输出
          const result = response.choices[0]?.message?.content;
          const usage = response.usage;
          console.log('Token 使用详情：', {
            prompt_tokens: usage?.prompt_tokens,
            completion_tokens: usage?.completion_tokens,
            total_tokens: usage?.total_tokens,
          });
          return result;
        }
      } catch (error) {
        lastError = error;
        console.error(`客户端 ${i + 1} 调用失败：`, error.message);

        // 如果是 401 或 429（权限/限流），马上切下一个 Key
        if (error.status === 401 || error.status === 429) {
          continue;
        }

        // 其他错误（如网络超时）也切 Key，但只切一次
        if (i === 0) {
          console.log('网络错误，尝试备用客户端...');
          continue;
        }

        // 所有重试都失败，抛出最终错误
        throw new Error(`所有客户端均失败。最后错误：${error.message}`);
      }
    }

    // 所有客户端都试完了还不行
    throw new Error(`所有客户端均失败。最后错误：${lastError?.message}`);
  });
}

// 导出
export { callDoubao };

## 第三步：封号防护——这 5 条一个都不能省

**1. 限流比你想的更重要。**

Doubao 企业 API 的限流阈值不是公开的。我用 `bottleneck` 库手动限流到每秒 10 个请求——这是我自己的经验值，低于这个值不容易触发限流，高于这个值就要自己试了。

**2. 不要把 Key 暴露给前端。**

有人图方便在前端代码里直接写 Key，或者通过客户端发请求——这就是自杀式操作。Key 一旦泄露，黑客就能用你的 Key 生成广告文案，最后账单全算在你头上。

**3. 每次调用都输出 Token 使用量。**

代码里有 `console.log('Token 使用详情：', ...)` 这一行，很多人觉得多余。但在企业场景里，这一行帮你监控每笔费用、发现异常调用。我靠这个抓到过一次 Key 被同事误用到其他项目去测大图生成——跑了一周，花了 200 多块钱。

**4. 请求超时和重试必须配。**

企业网络环境千奇百怪——有的公司用内网代理，有的机房对 HTTPS 有限制。不配超时，请求可能在队列里卡死几十秒，整个 Node 进程都被拖慢。上面代码里 `timeout: 60000` 是基于我踩坑后调出来的值。

**5. 生产环境禁止明文存储 Key。**

`.env` 文件不能提交到代码仓库。Git 里配 `.gitignore` 忽略 `.env`，用 CI/CD 的 Secrets 管理或环境变量注入。如果你用 Docker 部署，用 Docker Secrets 或者运行时挂载环境变量。

## 调试常见报错对照表

| 报错信息                               | 原因                              | 解决方案                                                   |
| ---------------------------------------- | ----------------------------------- | ------------------------------------------------------------ |
| `401 Unauthorized`                     | API Key 无效或过期                | 检查 `.env` 里的 Key，确认没有空格或换行，重新在千聚控制台生成 |
| `429 Too Many Requests`                | 限流，当前 Key 请求数超过阈值     | 降低并发，切换备用 Key，等待几秒再重试                       |
| `400 Bad Request - model not found`    | 模型名称错误或不支持              | 检查 `DOUBAO_MODEL` 名称，确认在千聚模型列表内              |
| `504 Gateway Timeout`                  | 请求超时，网络不通或服务端卡顿    | 增大 `timeout` 值，检查网络连通性，确认 `base_url` 正确     |
| `500 Internal Server Error`            | 服务端临时故障                    | 启用备用 Key 自动切换，代码里有兜底逻辑                      |
| `400 - invalid messages structure`     | 消息格式不符合 Doubao 规范        | 检查 messages 数组格式，确认 `role` 只能是 `system`/`user`/`assistant` |
| `ECONNREFUSED` 或 `ETIMEDOUT`          | 网络层面无法连接到 API 地址       | 检查 `base_url` 是否写对，确认公司防火墙是否允许 HTTPS 出站 |
